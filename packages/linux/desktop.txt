# Desktop — GNOME + GUI apps
# Requires multilib repo enabled for steam

# Desktop environment
gnome                    # full GNOME desktop (gnome-shell, GDM, Nautilus, etc.)
gnome-tweaks             # settings beyond the standard Settings app
xdg-desktop-portal-gnome # desktop portal (file picker, screen capture)
xdg-user-dirs            # standard home dirs (Documents, Downloads, etc.)

# Screenshot (supplements GNOME's built-in screenshot tool)
grim                     # screenshot tool (Wayland)
slurp                    # region selection for grim

# Clipboard
wl-clipboard             # wl-copy / wl-paste
cliphist                 # clipboard history manager

# Audio (likely pre-installed on CachyOS, ensure present)
pipewire
pipewire-alsa
pipewire-pulse
wireplumber
pavucontrol              # audio control GUI

# Qt Wayland support (needed for Qt apps to run natively on Wayland)
qt5-wayland
qt6-wayland

# Terminal
ghostty                  # using ghostty — already familiar from Mac

# Editors / IDEs (VSCode is in aur.txt — proprietary build)
zed                      # fast, modern editor with native Claude assistant

# Browser + general GUI
firefox

# Fonts
# (see aur.txt for Monaspace and Nerd Fonts)

# Media
mpv                      # video player (better Wayland support than VLC)
imv                      # image viewer (Wayland-native)

# Documents
zathura                  # keyboard-driven PDF viewer
zathura-pdf-mupdf        # PDF backend for zathura

# Bluetooth + networking
blueman                  # Bluetooth manager GUI
networkmanager           # network manager (likely pre-installed)
network-manager-applet   # tray icon for NetworkManager

# File manager (GNOME includes Nautilus; Thunar as lightweight alternative)
thunar

# Games
steam                    # requires multilib repo

# Keyboard
keyd                     # kernel-level key remapper (Super→Ctrl for mac-like shortcuts)

# Docker (native on Linux — no Colima needed)
docker
docker-compose
docker-buildx
